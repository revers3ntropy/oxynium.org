Oxynium

Oxynium is a statically typed, low level, compiled programming language

Key features include:

  • Powerful type system with generics
  • First-class functions and lambda functions
  • Macros
  • Classes
  • Operator overloading
  • Interoperability with other languages
  • Explicit error handling

Getting Started

Read the docs at oxynium.org/docs

Install for Linux / MacOS

curl -sSL https://oxynium.org/scripts/install | bash

start a new terminal session, and then

oxy hello_world.oxy \&\& ./oxy-out

Fibonacci Numbers

const N = 10
def fibonacci (n: Int) Int {
    if n \<= 1 -\>
        return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

print(fibonacci(N).Str()); // 55

FizzBuzz

def main() {
    for i in range(100) {
        print(fizzbuzz(i));
    }
}

def fizzbuzz (n: Int) Str {
    if n % 3 == 0 \&\& n % 5 == 0 {
        return "FizzBuzz";
    }
    if n % 3 == 0 -\> return "Fizz";
    if n % 5 == 0 -\> return "Buzz";
    return n.Str();
}

Queue class using Generics 

class LoggedQueue\<T\> {
	backing_list: List\<T\>,

	def make_empty\<S\>()
		-\> new LoggedQueue\<S\> { backing_list: List.empty!\<S\>() },

	def push(self, item: T) LoggedQueue\<T\> {
		self.backing_list.push(item);
		print("added item " + self.len().Str());
		return self;
	}

	def pop(self) T {
		if self.len() \< 1 -\> panic("pop on empty list");
		let val = self.backing_list.remove_at(0).unwrap();
		print("popped item " + self.len().Str());
		return val;
	}

	def len(self) -\> self.backing_list.len()
}

def main() {
	let q = LoggedQueue.make_empty!\<Int\>()
            	.push(5)
            	.push(6);

    print("process: " + q.pop().Str()); // 5
    print("then: " + q.pop().Str()); // 6
}
